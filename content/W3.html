# 從 URL 讀取學生資料
def fetch_student_data(url):
    try:
        response = urllib.request.urlopen(url)
        data = response.read()
        # 檢查資料類型，若為 bytes 則解碼
        if isinstance(data, bytes):
            data = data.decode('utf-8')
        return data.splitlines(), None
    except Exception as e:
        return None, f"無法讀取資料：{str(e)}"

# 解析學生資料並找出未分組學員
def find_ungrouped_students(data_lines):
    if not data_lines or len(data_lines) <= 1:
        return [], "學員名單讀取失敗或無有效資料"
    
    all_students = data_lines[1:]  # 跳過標題行
    total_count = len(all_students)
    ungrouped = []

    for line in all_students:
        parts = line.split()  # 用空格分割學號和帳號
        student_id = parts[0]  # 第一欄為學號
        account = parts[1] if len(parts) > 1 else student_id  # 若無帳號，用學號代替
        account = account.strip()

        # 若帳號不在已分組集合中，記錄學號
        if account not in GROUPED_ACCOUNTS:
            ungrouped.append(student_id)

    return ungrouped, f"總共有 {total_count} 名學員"

# 主程式
url = "https://mde.tw/list/2a.txt"
data_lines, error = fetch_student_data(url)

# 儲存輸出結果
output = []
if data_lines is not None:
    ungrouped, message = find_ungrouped_students(data_lines)
    output.append(message)
    output.append("未分組學員學號清單：")
    for student_id in ungrouped:
        output.append(student_id)
else:
    output.append(error)

# 在 CMSIMDE 中，print 會顯示在控制台，若需網頁顯示，需返回字串
for line in output:
    print(line)
