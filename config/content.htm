# 從 URL 讀取學生資料
def fetch_student_data(url):
    try:
        response = urllib.request.urlopen(url)
        data = response.read()
        # 檢查資料類型，若為 bytes 則解碼
        if isinstance(data, bytes):
            data = data.decode('utf-8')
        return data.splitlines(), None
    except Exception as e:
        return None, f"無法讀取資料：{str(e)}"

# 解析學生資料並找出未分組學員
def find_ungrouped_students(data_lines):
    if not data_lines or len(data_lines) <= 1:
        return [], "學員名單讀取失敗或無有效資料"
    
    all_students = data_lines[1:]  # 跳過標題行
    total_count = len(all_students)
    ungrouped = []

    for line in all_students:
        parts = line.split()  # 用空格分割學號和帳號
        student_id = parts[0]  # 第一欄為學號
        account = parts[1] if len(parts) > 1 else student_id  # 若無帳號，用學號代替
        account = account.strip()

        # 若帳號不在已分組集合中，記錄學號
        if account not in GROUPED_ACCOUNTS:
            ungrouped.append(student_id)

    return ungrouped, f"總共有 {total_count} 名學員"

# 主程式
url = "https://mde.tw/list/2a.txt"
data_lines, error = fetch_student_data(url)

# 儲存輸出結果
output = []
if data_lines is not None:
    ungrouped, message = find_ungrouped_students(data_lines)
    output.append(message)
    output.append("未分組學員學號清單：")
    for student_id in ungrouped:
        output.append(student_id)
else:
    output.append(error)

# 在 CMSIMDE 中，print 會顯示在控制台，若需網頁顯示，需返回字串
for line in output:
    print(line)

提6.
Webots 中建立一個簡單場景：一個機器人在平面上移動並避開障礙物。以下是手把手說明如何透過圖形介面完成：
步驟 1：啟動 Webots
  1. 打開 Webots 應用程式。
  2. 在啟動畫面選擇「New World」（新建世界）。
步驟 2：設置基本場景
  1. 加入地板
    • 在頂部選單點擊「Add」（添加）。
    • 選擇「Solid」（實體物件） > 「Box」（方塊）。
    • 在屬性面板中設置：
      • 大小（Size）：10m x 10m x 0.1m（寬廣的平面）。
      • 名稱（Name）：Floor。
    • 注意：確保地板位於 Z=0，避免懸浮。
  2. 加入機器人
    • 點擊「Add」 > 「Robot」（機器人）。
    • 在下拉選單中選擇「E-puck」（一個簡單的雙輪機器人）。
    • 調整位置（Translation）：例如 (0, 0, 0.05)，使其位於地板上。
    • 注意：檢查機器人是否與地板接觸，避免穿透或漂浮。
  3. 加入障礙物
    • 點擊「Add」 > 「Solid」 > 「Box」。
    • 設置大小：1m x 1m x 0.5m。
    • 位置（Translation）：(2, 0, 0.25)。
    • 名稱：Obstacle。
步驟 3：添加控制器
  1. 點擊機器人節點（E-puck）。
  2. 在屬性面板中找到「Controller」欄位。
  3. 點擊下拉選單，選擇「<extern>」（外部控制器）。
  4. 點擊「Wizard」（精靈），選擇「Create a new controller」。
  5. 輸入名稱（如「avoid_obstacle」），選擇語言（如 Python）。
  6. Webots 會生成一個簡單的控制器檔案。
步驟 4：編輯控制器
  1. 在控制器檔案中輸入簡單的避障邏輯（範例 Python 程式碼）：
    from controller import Robot

robot = Robot()
timestep = 64
left_motor = robot.getDevice("left wheel motor")
right_motor = robot.getDevice("right wheel motor")
left_motor.setPosition(float('inf'))
right_motor.setPosition(float('inf'))
distance_sensor = robot.getDevice("ds0")
distance_sensor.enable(timestep)

while robot.step(timestep) != -1:
    distance = distance_sensor.getValue()
    if distance < 950:  # 檢測到障礙物
        left_motor.setVelocity(-5)
        right_motor.setVelocity(5)  # 轉向
    else:
        left_motor.setVelocity(5)
        right_motor.setVelocity(5)  # 前進
    1. 保存檔案。
    步驟 5：運行模擬
    1. 點擊頂部選單的「Run」（運行按鈕）。
    2. 觀察機器人是否能移動並避開障礙物。
    3. 若有問題，檢查：
      • 機器人與地板的接觸是否正常。
      • 感測器是否啟用。
      • 控制器是否正確綁定。
    注意事項
    • 坐標系統：Webots 使用 X（左右）、Y（前後）、Z（上下）的坐標系，確保物件位置合理。
    • 物理屬性：檢查物件的質量與摩擦係數，避免異常滑動。
    • 控制器同步：確保控制器與機器人節點正確關聯。
